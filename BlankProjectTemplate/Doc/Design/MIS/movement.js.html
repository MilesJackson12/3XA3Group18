/**
 * A function used to move one of the players in the game. Also checks from collisions with walls and does not allow movement through walls.
 * @function
 * @param {int} playerID - ID of the player to be moved.
 * @param {String} dir - Direction to move the player.
 */
function moveplayers(dir, playerID){
    let row = player.row;
    let col = player.col;
    let row2 = player2.row;
    let col2 = player2.col;

    if (dir == "left") {
      col--;
    }
    // up arrow key
    else if (dir == "up") {
      row--;
    }
    // right arrow key
    else if (dir == "right") {
      col++;
    }
    // down arrow key
    else if (dir == "down") {
      row++;
    }
  
     // left arrow key
    else if (dir == "left2") {
      col2--;
    }
    // up arrow key
    else if (dir == "up2") {
      row2--;
    }
    // right arrow key
    else if (dir == "right2") {
      col2++;
    }
    // down arrow key
    else if (dir == "down2") {
      row2++;
    }
    

    if (!cells[row][col]) {
        player.row = row;
        player.col = col;
    }
    if (!cells[row2][col2]) {
      player2.row = row2;
      player2.col = col2;
    }
}
